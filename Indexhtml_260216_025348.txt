  <script>
    const app = {
      data: {
        storeActive: false, 
        posts: [
          { id: 1, user: "Thomas Beaver", text: "Welcome to Tomsminimall Freedom Platform. The world is now open for business. No fees, no borders.", time: "Pinned" }
        ],
        products: [
          { id: 1, name: "Global License", price: 30, stock: 1000, seller: "Tomsminimall" },
          { id: 2, name: "Organic Coffee", price: 18, stock: 45, seller: "Colombia Direct" }
        ]
      },

      init() {
        this.load();
        this.renderFeed();
        this.renderMall();
        this.checkStoreStatus();
      },

      load() {
        const d = localStorage.getItem('tm_world_v1');
        if(d) this.data = JSON.parse(d);
      },

      save() {
        localStorage.setItem('tm_world_v1', JSON.stringify(this.data));
      },

      switchTab(tab) {
        document.querySelectorAll('.tab-content').forEach(el => el.classList.add('hidden'));
        document.getElementById('tab-' + tab).classList.remove('hidden');
        document.querySelectorAll('nav button').forEach(b => { b.classList.remove('nav-active', 'text-slate-500'); b.classList.add('text-slate-500'); });
        document.getElementById('nav-' + tab).classList.add('nav-active');
        document.getElementById('nav-' + tab).classList.remove('text-slate-500');
      },

      checkStoreStatus() {
        const locked = document.getElementById('store-locked');
        const unlocked = document.getElementById('store-unlocked');
        if (this.data.storeActive) {
          locked.classList.add('hidden');
          unlocked.classList.remove('hidden');
          this.renderInventory();
        } else {
          locked.classList.remove('hidden');
          unlocked.classList.add('hidden');
        }
      },

      activateStore() {
        if(confirm("Launch Worldwide Storefront?\nCost: $30.00 (One Time)")) {
          const btn = document.querySelector('#store-locked button');
          btn.innerText = "Connecting to Global Network...";
          setTimeout(() => {
            alert("Success! Your store is now active on the global node.");
            this.data.storeActive = true;
            this.save();
            this.checkStoreStatus();
          }, 1500);
        }
      },

      stockShelf() {
        const name = document.getElementById('prodName').value;
        const price = document.getElementById('prodPrice').value;
        const stock = document.getElementById('prodStock').value;
        if(!name || !price) return alert("Product details required.");
        
        this.data.products.unshift({ id: Date.now(), name, price, stock: stock || 1, seller: "My Global Store" });
        this.save();
        this.renderMall(); 
        this.renderInventory();
        document.getElementById('prodName').value = '';
        document.getElementById('prodPrice').value = '';
        document.getElementById('prodStock').value = '';
      },

      renderMall() {
        const el = document.getElementById('mall-grid');
        el.innerHTML = this.data.products.map(p => `
          <div class="glass-card p-5 rounded-[1.5rem] relative overflow-hidden group hover:bg-white/10 transition">
            <div class="absolute top-0 right-0 bg-blue-600 text-white text-[9px] font-bold px-3 py-1 rounded-bl-xl uppercase tracking-widest shadow-lg">Global Shipping</div>
            <div class="text-4xl mb-3 mt-2">📦</div>
            <h3 class="font-bold text-white text-lg leading-tight mb-1 truncate">${p.name}</h3>
            <p class="text-[10px] text-slate-400 uppercase font-bold mb-4">Sold by: ${p.seller}</p>
            <div class="flex justify-between items-end border-t border-white/10 pt-3">
              <div><span class="text-[9px] text-slate-400 block uppercase font-bold">USD</span><span class="text-xl font-black text-emerald-400">$${p.price}</span></div>
              <button class="bg-white text-slate-900 px-4 py-2 rounded-lg font-bold text-[10px] uppercase hover:bg-emerald-400 transition shadow-lg">Purchase</button>
            </div>
          </div>
        `).join('');
      },

      renderInventory() {
        const el = document.getElementById('my-inventory');
        const myItems = this.data.products.filter(p => p.seller === "My Global Store");
        el.innerHTML = myItems.length ? myItems.map(p => `
          <div class="glass-card p-4 rounded-xl flex justify-between items-center border border-white/5">
            <div><h4 class="font-bold text-white text-sm">${p.name}</h4><p class="text-[10px] text-slate-400">Qty: ${p.stock} | $${p.price}</p></div>
            <button onclick="app.removeItem(${p.id})" class="text-red-400 hover:scale-110 transition p-2">🗑️</button>
          </div>
        `).join('') : `<div class="text-center py-8 text-slate-500 text-sm italic">Shelf Empty. Upload Inventory.</div>`;
      },

      removeItem(id) {
        if(confirm("Delete item from Global Mall?")) {
          this.data.products = this.data.products.filter(p => p.id !== id);
          this.save();
          this.renderMall();
          this.renderInventory();
        }
      },

      renderFeed() {
        const el = document.getElementById('feed-stream');
        el.innerHTML = this.data.posts.map(p => `
          <div class="glass-card p-5 rounded-2xl shadow-lg fade-in border-l-4 border-blue-600">
            <div class="flex justify-between items-start mb-2"><span class="text-xs font-black text-blue-400 uppercase">@${p.user}</span><span class="text-[9px] text-slate-500 font-bold uppercase">${p.time}</span></div>
            <p class="text-sm text-slate-200 leading-relaxed font-medium">${p.text}</p>
          </div>
        `).join('');
      },

      addPost() {
        const input = document.getElementById('postInput');
        if(!input.value.trim()) return;
        this.data.posts.unshift({ id: Date.now(), user: "Global_Trader", text: input.value, time: "Just Now" });
        this.save();
        this.renderFeed();
        input.value = '';
      },

      resetData() {
        if(confirm("Reset Local Data cache?")) { localStorage.removeItem('tm_world_v1'); location.reload(); }
      }
    };
    app.init();
  </script>

            </button>
          </div>
        </div>

        <h3 class="font-bold text-slate-500 text-xs uppercase tracking-widest mb-4 ml-2">My Global Inventory</h3>
        <div id="my-inventory" class="space-y-3"></div>
      </div>
    </div>

  </main>

  <nav class="fixed bottom-0 left-0 right-0 glass-nav px-8 pb-8 pt-4 z-50">
    <div class="max-w-xl mx-auto flex justify-between items-center">
      <button onclick="app.switchTab('feed')" id="nav-feed" class="flex flex-col items-center gap-1 nav-active w-1/3">
        <i class="u-icon">📡</i>
        <span class="text-[10px] font-black uppercase tracking-widest">Feed</span>
      </button>
      <button onclick="app.switchTab('mall')" id="nav-mall" class="flex flex-col items-center gap-1 text-slate-500 w-1/3">
        <i class="u-icon">🌏</i>
        <span class="text-[10px] font-black uppercase tracking-widest">World Mall</span>
      </button>
      <button onclick="app.switchTab('store')" id="nav-store" class="flex flex-col items-center gap-1 text-slate-500 w-1/3">
        <i class="u-icon">🏪</i>
        <span class="text-[10px] font-black uppercase tracking-widest">My Store</span>

  <script>
    const app = {
      data: {
        storeActive: false, 
        posts: [
          { id: 1, user: "Thomas Beaver", text: "Welcome to Tomsminimall Freedom Platform. The world is now open for business. No fees, no borders.", time: "Pinned" }
        ],
        products: [
          { id: 1, name: "Global License", price: 30, stock: 1000, seller: "Tomsminimall" },
          { id: 2, name: "Organic Coffee", price: 18, stock: 45, seller: "Colombia Direct" }
        ]
      },

      init() {
        this.load();
        this.renderFeed();
        this.renderMall();
        this.checkStoreStatus();
      },

      load() {
        const d = localStorage.getItem('tm_world_v1');
        if(d) this.data = JSON.parse(d);
      },

      save() {
        localStorage.setItem('tm_world_v1', JSON.stringify(this.data));
      },

      switchTab(tab) {
        document.querySelectorAll('.tab-content').forEach(el => el.classList.add('hidden'));
        document.getElementById('tab-' + tab).classList.remove('hidden');
        document.querySelectorAll('nav button').forEach(b => { b.classList.remove('nav-active', 'text-slate-500'); b.classList.add('text-slate-500'); });
        document.getElementById('nav-' + tab).classList.add('nav-active');
        document.getElementById('nav-' + tab).classList.remove('text-slate-500');
      },

      checkStoreStatus() {
        const locked = document.getElementById('store-locked');
        const unlocked = document.getElementById('store-unlocked');
        if (this.data.storeActive) {
          locked.classList.add('hidden');
          unlocked.classList.remove('hidden');
          this.renderInventory();
        } else {
          locked.classList.remove('hidden');
          unlocked.classList.add('hidden');
        }
      },

      activateStore() {
        if(confirm("Launch Worldwide Storefront?\nCost: $30.00 (One Time)")) {
          const btn = document.querySelector('#store-locked button');
          btn.innerText = "Connecting to Global Network...";
          setTimeout(() => {
            alert("Success! Your store is now active on the global node.");
            this.data.storeActive = true;
            this.save();
            this.checkStoreStatus();
          }, 1500);
        }
      },

      stockShelf() {
        const name = document.getElementById('prodName').value;
        const price = document.getElementById('prodPrice').value;
        const stock = document.getElementById('prodStock').value;
        if(!name || !price) return alert("Product details required.");
        
        this.data.products.unshift({ id: Date.now(), name, price, stock: stock || 1, seller: "My Global Store" });
        this.save();
        this.renderMall(); 
        this.renderInventory();
        document.getElementById('prodName').value = '';
        document.getElementById('prodPrice').value = '';
        document.getElementById('prodStock').value = '';
      },

      renderMall() {
        const el = document.getElementById('mall-grid');
        el.innerHTML = this.data.products.map(p => `
          <div class="glass-card p-5 rounded-[1.5rem] relative overflow-hidden group hover:bg-white/10 transition">
            <div class="absolute top-0 right-0 bg-blue-600 text-white text-[9px] font-bold px-3 py-1 rounded-bl-xl uppercase tracking-widest shadow-lg">Global Shipping</div>
            <div class="text-4xl mb-3 mt-2">📦</div>
            <h3 class="font-bold text-white text-lg leading-tight mb-1 truncate">${p.name}</h3>
            <p class="text-[10px] text-slate-400 uppercase font-bold mb-4">Sold by: ${p.seller}</p>
            <div class="flex justify-between items-end border-t border-white/10 pt-3">
              <div><span class="text-[9px] text-slate-400 block uppercase font-bold">USD</span><span class="text-xl font-black text-emerald-400">$${p.price}</span></div>
              <button class="bg-white text-slate-900 px-4 py-2 rounded-lg font-bold text-[10px] uppercase hover:bg-emerald-400 transition shadow-lg">Purchase</button>
            </div>
          </div>
        `).join('');
      },

      renderInventory() {
        const el = document.getElementById('my-inventory');
        const myItems = this.data.products.filter(p => p.seller === "My Global Store");
        el.innerHTML = myItems.length ? myItems.map(p => `
          <div class="glass-card p-4 rounded-xl flex justify-between items-center border border-white/5">
            <div><h4 class="font-bold text-white text-sm">${p.name}</h4><p class="text-[10px] text-slate-400">Qty: ${p.stock} | $${p.price}</p></div>
            <button onclick="app.removeItem(${p.id})" class="text-red-400 hover:scale-110 transition p-2">🗑️</button>
          </div>
        `).join('') : `<div class="text-center py-8 text-slate-500 text-sm italic">Shelf Empty. Upload Inventory.</div>`;
      },

      removeItem(id) {
        if(confirm("Delete item from Global Mall?")) {
          this.data.products = this.data.products.filter(p => p.id !== id);
          this.save();
          this.renderMall();
          this.renderInventory();
        }
      },

      renderFeed() {
        const el = document.getElementById('feed-stream');
        el.innerHTML = this.data.posts.map(p => `
          <div class="glass-card p-5 rounded-2xl shadow-lg fade-in border-l-4 border-blue-600">
            <div class="flex justify-between items-start mb-2"><span class="text-xs font-black text-blue-400 uppercase">@${p.user}</span><span class="text-[9px] text-slate-500 font-bold uppercase">${p.time}</span></div>
            <p class="text-sm text-slate-200 leading-relaxed font-medium">${p.text}</p>
          </div>
        `).join('');
      },

      addPost() {
        const input = document.getElementById('postInput');
        if(!input.value.trim()) return;
        this.data.posts.unshift({ id: Date.now(), user: "Global_Trader", text: input.value, time: "Just Now" });
        this.save();
        this.renderFeed();
        input.value = '';
      },

      resetData() {
        if(confirm("Reset Local Data cache?")) { localStorage.removeItem('tm_world_v1'); location.reload(); }
      }
    };
    app.init();
  </script>
</body>
